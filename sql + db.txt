CREATE TABLE PersonInformation (
    RollNo INT PRIMARY KEY,
    Name VARCHAR(100),
    Age INT,
    DateOfBirth DATE,
    Gender CHAR(1) -- M or F or others
);


CREATE TABLE ProjectInformation (
    ProjectID INT PRIMARY KEY,
    ProjectName VARCHAR(100),
    Subject VARCHAR(100),
    Status VARCHAR(20) -- 'accepted' or 'not accepted'
);

CREATE TABLE PersonProject (
    RollNo INT,
    ProjectID INT,
    PRIMARY KEY (RollNo, ProjectID),
    FOREIGN KEY (RollNo) REFERENCES PersonInformation(RollNo),
    FOREIGN KEY (ProjectID) REFERENCES ProjectInformation(ProjectID)
);
